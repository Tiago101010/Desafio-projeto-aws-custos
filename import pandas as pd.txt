import pandas as pd
import requests
import json
import openai

users = [
    {'id': 1, 'name': 'Naruto', 'news': []},
    {'id': 2, 'name': 'Hinata', 'news': []},
    {'id': 2, 'name': 'Sasuke', 'news': []}
]

openai_api_key = "SUA_CHAVE_AQUI" # Substitua pela sua chave da API real
client = openai.OpenAI(api_key=openai_api_key)

def generate_ai_news (user):
  completions = client.chat.completions.create(
      model="gpt-5.2", # ou "gpt-4" se "gpt-5.2" não estiver disponível ou preferir
      messages=[
      {
          "role":"system",
          "content":"Voce é um especialista em marketing bancário."
      },
      {
          "role":"user",
          "content":f"Crie uma mensagem para {user['name']} sobre a importância dos investimentos (máximo de 100 caracteres)"
      },
    ]
  )
  responseChatGpt = completions.choices[0].message.content
  return responseChatGpt

for user in users:
  news = generate_ai_news(user)
  print(f"Mensagem para {user['name']}: {news}")
